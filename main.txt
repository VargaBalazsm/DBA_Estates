// Inicializálás - LocalStorage ellenőrzés és beállítás
if (!localStorage.getItem('ingatlanok')) {
    localStorage.setItem('ingatlanok', JSON.stringify(initialData));
}

// Ingatlan szolgáltatás objektum
const ingatlanService = {
    // Összes ingatlan lekérdezése (READ)
    async getAll() {
        try {
            // Szimulált API késleltetés
            await new Promise(resolve => setTimeout(resolve, 300));
            return JSON.parse(localStorage.getItem('ingatlanok')) || [];
        } catch (error) {
            console.error('Hiba az adatok betöltésekor:', error);
            return [];
        }
    },

    // Egy ingatlan lekérdezése ID alapján
    async getById(id) {
        const ingatlanok = await this.getAll();
        return ingatlanok.find(i => i.id === id);
    },

    // Új ingatlan létrehozása (CREATE)
    async create(ingatlan) {
        try {
            const ingatlanok = await this.getAll();
            const newId = ingatlanok.length > 0 ? Math.max(...ingatlanok.map(i => i.id)) + 1 : 1;
            
            const ujIngatlan = {
                ...ingatlan,
                id: newId
            };
            
            ingatlanok.push(ujIngatlan);
            localStorage.setItem('ingatlanok', JSON.stringify(ingatlanok));
            
            return ujIngatlan;
        } catch (error) {
            console.error('Hiba a létrehozáskor:', error);
            throw new Error('Nem sikerült menteni az ingatlant!');
        }
    },

    // Ingatlan törlése (DELETE)
    async delete(id) {
        try {
            let ingatlanok = await this.getAll();
            const eredetiHossz = ingatlanok.length;
            
            ingatlanok = ingatlanok.filter(i => i.id !== id);
            
            if (ingatlanok.length === eredetiHossz) {
                throw new Error('Az ingatlan nem található!');
            }
            
            localStorage.setItem('ingatlanok', JSON.stringify(ingatlanok));
            return true;
        } catch (error) {
            console.error('Hiba a törléskor:', error);
            throw error;
        }
    },

    // Szimulált Fetch API hívás (JSON fájl szimuláció)
    async fetchFromJSON() {
        // Ez a függvény szimulálja, mintha egy külső JSON fájlból töltenénk
        // Valós környezetben: fetch('data/ingatlanok.json')
        return new Promise((resolve) => {
            setTimeout(() => {
                resolve([...initialData]);
            }, 500);
        });
    }
};

// HTML template generátor ingatlan kártyákhoz
function ingatlanCardTemplate(ingatlan, showDelete = false) {
    const formatAr = new Intl.NumberFormat('hu-HU', {
        style: 'currency',
        currency: 'HUF',
        maximumFractionDigits: 0
    }).format(ingatlan.ar);

    const formatTerulet = `${ingatlan.alapterulet} m²`;
    const szobaText = ingatlan.szobak > 0 ? `${ingatlan.szobak} szoba` : 'N/A';

    return `
        <div class="col-md-6 col-lg-4">
            <div class="card ingatlan-card h-100">
                <div class="image-container">
                    <img src="${ingatlan.kep}" class="card-img-top" alt="${ingatlan.cim}">
                    <span class="tipus-badge">${ingatlan.tipus.toUpperCase()}</span>
                    ${showDelete ? `<button class="delete-btn" onclick="torlesModal(${ingatlan.id})" title="Törlés">
                        <i data-lucide="trash-2" style="width: 18px; height: 18px;"></i>
                    </button>` : ''}
                </div>
                <div class="card-body">
                    <h5 class="card-title">${ingatlan.cim}</h5>
                    <p class="text-muted mb-2"><i data-lucide="map-pin" style="width: 16px; height: 16px; display: inline; vertical-align: text-bottom;"></i> ${ingatlan.telepules}</p>
                    <div class="ar-badge">${formatAr}</div>
                    <div class="property-meta">
                        <span class="property-meta-item">
                            <i data-lucide="maximize"></i> ${formatTerulet}
                        </span>
                        <span class="property-meta-item">
                            <i data-lucide="bed"></i> ${szobaText}
                        </span>
                    </div>
                    <p class="card-text">${ingatlan.leiras.substring(0, 100)}${ingatlan.leiras.length > 100 ? '...' : ''}</p>
                    <button class="btn btn-primary w-100 mt-auto">Részletek</button>
                </div>
            </div>
        </div>
    `;
}
